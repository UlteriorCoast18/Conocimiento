\def\section#1{%
  \par\bigskip\bigskip
  \hrule\par\nobreak\noindent
  \refstepcounter{section}%
  \addcontentsline{toc}{chapter}{#1}%
  \reset@font
  { \large \scshape
    \strut\S \thesection \quad
    #1}% 
    \hrule   
  \par
  \medskip
}

\def\subsection#1{%
  \par\bigskip\bigskip
  \hrule\par\nobreak\noindent
  \refstepcounter{subsection}%
  \addcontentsline{toc}{section}{#1}%
  \reset@font
  { \normalsize \scshape
    \strut\S \thesubsection \quad
    #1}% 
    \hrule   
  \par
  \medskip
}
